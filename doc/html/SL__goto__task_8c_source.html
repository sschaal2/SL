549 \note calculate_min_jerk_next_step
550 \date August 1994
551 
552 \remarks
553 
554 Given the time to go, the desired state is updated according to the min jerk
555 formula.
556 
557  *******************************************************************************
558  Function Parameters: [in]=input,[out]=output
559 
560  \param[in,out] state : the current state
561  \param[in] goal : the desired state
562  \param[in] tau : the desired movement duration until the goal is
563 
564  ******************************************************************************/
565 static int
566 calculate_min_jerk_next_step (SL_DJstate *state, SL_DJstate *goal,double tau)
567 
568 {
569  double t1,t2,t3,t4,t5;
570  double tau1,tau2,tau3,tau4,tau5;
571  int i,j;
572  double delta_t;
573 
574  delta_t = 1./((double)task_servo_rate);
575 
576  if (delta_t > tau || delta_t <= 0) {
577  return FALSE;
578  }
579 
580  t1 = delta_t;
581  t2 = t1 * delta_t;
582  t3 = t2 * delta_t;
583  t4 = t3 * delta_t;
584  t5 = t4 * delta_t;
585 
586  tau1 = tau;
587  tau2 = tau1 * tau;
588  tau3 = tau2 * tau;
589  tau4 = tau3 * tau;
590  tau5 = tau4 * tau;
591 
592  for (j=1; j<=n_dofs; ++j) {
593 
594  /* calculate the constants */
595  const double dist = goal[j].th - state[j].th;
596  const double p1 = goal[j].th;
597  const double p0 = state[j].th;
598  const double a1t2 = goal[j].thdd;
599  const double a0t2 = state[j].thdd;
600  const double v1t1 = goal[j].thd;
601  const double v0t1 = state[j].thd;
602 
603  const double c1 = 6.*dist/tau5 + (a1t2 - a0t2)/(2.*tau3) -
604  3.*(v0t1 + v1t1)/tau4;
605  const double c2 = -15.*dist/tau4 + (3.*a0t2 - 2.*a1t2)/(2.*tau2) +
606  (8.*v0t1 + 7.*v1t1)/tau3;
607  const double c3 = 10.*dist/tau3+ (a1t2 - 3.*a0t2)/(2.*tau) -
608  (6.*v0t1 + 4.*v1t1)/tau2;
609  const double c4 = state[j].thdd/2.;
610  const double c5 = state[j].thd;
611  const double c6 = state[j].th;
612 
613  state[j].th = c1*t5 + c2*t4 + c3*t3 + c4*t2 + c5*t1 + c6;
614  state[j].thd = 5.*c1*t4 + 4*c2*t3 + 3*c3*t2 + 2*c4*t1 + c5;
615  state[j].thdd = 20.*c1*t3 + 12.*c2*t2 + 6.*c3*t1 + 2.*c4;
