212 \param[in,out] lstate : the desired state
213 \param[in] endeff : the endeffector parameters
214 \param[in] cbase : the position state of the base
215 \param[in] obase : the orientational state of the base
216 \param[in] ux : the external forces acting on each joint, in world

218 \param[out] fbase : the force/torque vector of the base in world coordinates

222 void SL_invDynNE::
223 SL_InvDynNEGeneral(SL_Jstate *cstate,SL_DJstate *lstate,SL_endeff *leff,
224  SL_Cstate *cbase, SL_quat *obase, SL_uext *ux,
225  double *fbase)
226 
227 {
228  int i;
229 
230  // this makes the arguments global variables
231  eff = leff;
232  basec = cbase;
233  baseo = obase;
234  uex = ux;
235 
236  // create a mixed desired/current state for proper inverse dynamics
237  for (i=1; i<=N_DOFS; ++i) {
238  state[i] = lstate[i];
239  if (cstate != NULL) {
240  state[i].th = cstate[i].th;
241  state[i].thd = cstate[i].thd;
242  }
243  }
244 
245 #include "InvDynNE_math.h"
246 
247  // add the friction term
248  for (i=1; i<=N_DOFS; ++i) {
249  SL_Jstate jt;
250  jt.th = state[i].th;
251  jt.thd = state[i].thd;
252  state[i].uff += compute_independent_joint_forces(jt,links[i]);
253  if (i > N_DOFS-N_DOFS_EST_SKIP)
254  state[i].uff = 0.0;
255  lstate[i].uff = state[i].uff;
256  }
