380

383  */

403

542  *******************************************************************************
543 \note compute_kinematics
544 \date June 99
545 
546 \remarks
547 
548  computes kinematic variables
549 
550  *******************************************************************************
551  Function Parameters: [in]=input,[out]=output
552 
553  none
554 
555  ******************************************************************************/
556 static void
557 compute_kinematics(void)
558 {
559  int i,j,r,o;
560  static int firsttime = TRUE;
561  static SL_DJstate *temp;
562  static Matrix last_J;
563  static Matrix last_Jbase;
564 
565  if (firsttime) {
566  temp=(SL_DJstate *)my_calloc((unsigned long)(n_dofs+1),sizeof(SL_DJstate),MY_STOP);
567  last_J = my_matrix(1,6*n_endeffs,1,n_dofs);
568  last_Jbase = my_matrix(1,6*n_endeffs,1,2*N_CART);
569  }
570 
571  /* compute the desired link positions */
572  linkInformationDes(joint_des_state,&base_state,&base_orient,endeff,
573  joint_cog_mpos_des,joint_axis_pos_des,joint_origin_pos_des,
574  link_pos_des,Alink_des,Adof_des);
575 
576  /* the desired endeffector information */
577  for (i=1; i<=N_CART; ++i) {
578  for (j=1; j<=n_endeffs; ++j) {
579  cart_des_state[j].x[i] = link_pos_des[link2endeffmap[j]][i];
580  }
581  }
582 
583  /* the desired quaternian of the endeffector */
584  for (j=1; j<=n_endeffs; ++j) {
585  linkQuat(Alink_des[link2endeffmap[j]],&(cart_des_orient[j]));
586  }
587 
588 
589  /* addititional link information */
590  linkInformation(joint_state,&base_state,&base_orient,endeff,
591  joint_cog_mpos,joint_axis_pos,joint_origin_pos,
592  link_pos,Alink,Adof);
593 
594  /* create the endeffector information */
595  for (i=1; i<=N_CART; ++i) {
596  for (j=1; j<=n_endeffs; ++j) {
597  cart_state[j].x[i] = link_pos[link2endeffmap[j]][i];
598  }
599  }
600 
601  /* the quaternian of the endeffector */
602  for (j=1; j<=n_endeffs; ++j) {
603  linkQuat(Alink[link2endeffmap[j]],&(cart_orient[j]));
604  }
605 
606  /* the COG position */
607  compute_cog();
608 
609  /* the jacobian */
610  jacobian(link_pos,joint_origin_pos,joint_axis_pos,J);
611  baseJacobian(link_pos,joint_origin_pos,joint_axis_pos,Jbase);
612 
613  jacobian(link_pos_des,joint_origin_pos_des,joint_axis_pos_des,Jdes);
614  baseJacobian(link_pos_des,joint_origin_pos_des,joint_axis_pos_des,Jbasedes);
615 
616  /* numerical time derivative of Jacobian */
617  if (!firsttime) {
618  mat_sub(J,last_J,dJdt);
619  mat_mult_scalar(dJdt,(double)ros_servo_rate,dJdt);
620  mat_sub(Jbase,last_Jbase,dJbasedt);
621  mat_mult_scalar(dJbasedt,(double)ros_servo_rate,dJbasedt);
622  }
623  mat_equal(J,last_J);
624  mat_equal(Jbase,last_Jbase);
625 
626  /* compute the cartesian velocities and accelerations */
627 
628  for (j=1; j<=n_endeffs; ++j) {
629 
630  for (i=1; i<=N_CART; ++i) {
631 
632  cart_state[j].xd[i] = 0.0;
633  cart_state[j].xdd[i] = 0.0;
634 
635  cart_orient[j].ad[i] = 0.0;
636  cart_orient[j].add[i] = 0.0;
637 
638  cart_des_state[j].xd[i] = 0.0;
639  cart_des_orient[j].ad[i] = 0.0;
640 
641  /* contributations from the joints */
642  for (r=1; r<=n_dofs; ++r) {
643  cart_state[j].xd[i] += J[(j-1)*6+i][r] * joint_state[r].thd;
644  cart_orient[j].ad[i] += J[(j-1)*6+i+3][r] * joint_state[r].thd;
645 
646  cart_des_state[j].xd[i] += Jdes[(j-1)*6+i][r] *joint_des_state[r].thd;
647  cart_des_orient[j].ad[i]+= Jdes[(j-1)*6+i+3][r] * joint_des_state[r].thd;
648 
649  cart_state[j].xdd[i] += J[(j-1)*6+i][r] * joint_state[r].thdd +
650  dJdt[(j-1)*6+i][r] * joint_state[r].thd;
651  cart_orient[j].add[i] += J[(j-1)*6+i+3][r] * joint_state[r].thdd +
652  dJdt[(j-1)*6+i+3][r] * joint_state[r].thd;
653  }
654 
655  /* contributations from the base */
656  for (r=1; r<=N_CART; ++r) {
657  cart_state[j].xd[i] += Jbase[(j-1)*6+i][r] * base_state.xd[r];
658  cart_orient[j].ad[i] += Jbase[(j-1)*6+i+3][r] * base_state.xd[r];
659 
660  cart_state[j].xd[i] += Jbase[(j-1)*6+i][3+r] * base_orient.ad[r];
661  cart_orient[j].ad[i] += Jbase[(j-1)*6+i+3][3+r] * base_orient.ad[r];
662 
663  cart_des_state[j].xd[i] += Jbasedes[(j-1)*6+i][r] * base_state.xd[r];
664  cart_des_orient[j].ad[i] += Jbasedes[(j-1)*6+i+3][r] * base_state.xd[r];
665 
666  cart_des_state[j].xd[i] += Jbasedes[(j-1)*6+i][3+r] * base_orient.ad[r];
667  cart_des_orient[j].ad[i] += Jbasedes[(j-1)*6+i+3][3+r] * base_orient.ad[r];
668 
669  cart_state[j].xdd[i] += Jbase[(j-1)*6+i][r] * base_state.xdd[r] +
670  dJbasedt[(j-1)*6+i][r] * base_state.xd[r];
671  cart_orient[j].add[i] += Jbase[(j-1)*6+i+3][r] * base_state.xdd[r] +
672  dJbasedt[(j-1)*6+i+3][r] * base_state.xd[r];
673 
674  cart_state[j].xdd[i] += Jbase[(j-1)*6+i][3+r] * base_orient.add[r] +
675  dJbasedt[(j-1)*6+i][3+r] * base_orient.ad[r];
676  cart_orient[j].add[i] += Jbase[(j-1)*6+i+3][3+r] * base_orient.add[r] +
677  dJbasedt[(j-1)*6+i+3][3+r] * base_orient.ad[r];
678  }
679 
680  }
681 
682  /* compute quaternion derivatives */
683  quatDerivatives(&(cart_orient[j]));
684  quatDerivatives(&(cart_des_orient[j]));
685  for (r=1; r<=N_QUAT; ++r)
686  cart_des_orient[j].qdd[r] = 0.0; // we don't have dJdes_dt so far
687 
688  }
689 
690  /* reset first time flag */
691  firsttime = FALSE;
692 
693 }
