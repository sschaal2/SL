1440  as cos(alpha/2) is thus never negative.

1452

1454  T = sqrt(T+1.0);
1455  q_aux[1] = 0.5*T;
1456  T = 0.5/T;
1457  q_aux[2] = ( R[3][2] - R[2][3] ) * T;
1458  q_aux[3] = ( R[1][3] - R[3][1] ) * T;
1459  q_aux[4] = ( R[2][1] - R[1][2] ) * T;
1460

1471  q_aux[i+2] = 0.5 * T;

1473  q_aux[1] = (R[k+1][j+1]-R[j+1][k+1])*T;
1474  q_aux[j+2] = (R[j+1][i+1]+R[i+1][j+1])*T;
