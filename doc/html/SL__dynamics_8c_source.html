259 /*!*****************************************************************************
260  *******************************************************************************
261 \note compute_independent_joint_forces
262 \date Sept 2010
263 
264 \remarks
265 
266 computes the generalized joint forces due to friction and spring terms, i.e.,
267 the sum of all forces that act per joint independently of all others. The sign
268 of the terms is as if they were on the LEFT side of the RBD equations:
269 
270 M qdd + C qd + G + f = u
271 
272  *******************************************************************************
273  Function Parameters: [in]=input,[out]=output
274 
275  \param[in] state : the joint state of the robot
276  \param[in] li : the link parameters for this joint
277 
278  returns the generalized joint force for this joint due friction and spring terms
279 
280  ******************************************************************************/
281 double
282 compute_independent_joint_forces(SL_Jstate state, SL_link li)
283 {
284  double f=0;
285 
286  f = state.thd*li.vis +
287  COULOMB_FUNCTION(state.thd,coulomb_slope)*li.coul +
288  state.th*li.stiff +
289  li.cons;
290 
291  return f;
292 }
