901 /*!*****************************************************************************
902 *******************************************************************************
903 \note receive_des_commands
904 \date Nov. 2007
905 
906 \remarks
907 
908 reads the commands from the joint_sim_state shared memory
909 structure
910 
911 
912 *******************************************************************************
913 Function Parameters: [in]=input,[out]=output
914 
915 none
916 
917 ******************************************************************************/
918 int
919 receive_des_commands(void)
920 {
921 
922  int i;
923  extern double *controller_gain_th;
924  extern double *controller_gain_thd;
925 
926 
927  if (semTake(sm_des_commands_sem,ns2ticks(TIME_OUT_NS)) == ERROR) {
928 
929  ++simulation_servo_errors;
930  return FALSE;
931 
932  }
933 
934  for (i=1; i<=n_dofs; ++i) {
935  joint_sim_state[i].u = (double) sm_des_commands->des_command[i].u -
936  (double) sm_des_commands->des_command[i].upd;
937  joint_des_state[i].th = (double) sm_des_commands->des_command[i].th;
938  joint_des_state[i].thd = (double) sm_des_commands->des_command[i].thd;
939  }
940 
941  // re-generate and add the PD command from the motor servo, as the
942  // delay between motor servo and simulation servo can cause instabilities
943  // in case of stiff contacts -- the reason is that the state info on the
944  // motor servo is one tick delayed; the old PD command has been subtracted
945  // from the total command above
946 
947  for (i=1; i<=n_dofs; ++i) {
948  if (sm_des_commands->des_command[i].upd != 0.0) {
949  joint_sim_state[i].u += (joint_des_state[i].th - joint_sim_state[i].th) *
950  controller_gain_th[i];
951  joint_sim_state[i].u += (joint_des_state[i].thd - joint_sim_state[i].thd) *
952  controller_gain_thd[i];
953  }
954 
955  if (fabs(joint_sim_state[i].u) > u_max[i]) {
956  joint_sim_state[i].u = macro_sign(joint_sim_state[i].u) * u_max[i];
957  }
958 
959  }
960 
961  semGive(sm_des_commands_sem);
962 
963  return TRUE;
964 }
