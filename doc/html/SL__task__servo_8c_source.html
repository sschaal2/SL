950 /*!*****************************************************************************

958 
959  *******************************************************************************
960  Function Parameters: [in]=input,[out]=output
961 
962  none
963 
964  ******************************************************************************/
965 static void
966 compute_kinematics(void)
967 {
968  int i,j,r,o;
969  static int firsttime = TRUE;
970  static SL_DJstate *temp;
971  static Matrix last_J;
972  static Matrix last_Jbase;
973 
974  if (firsttime) {
975  temp=(SL_DJstate *)my_calloc((unsigned long)(n_dofs+1),sizeof(SL_DJstate),MY_STOP);
976  last_J = my_matrix(1,6*n_endeffs,1,n_dofs);
977  last_Jbase = my_matrix(1,6*n_endeffs,1,2*N_CART);
978  }
979 
980  /* compute the desired link positions */
981  linkInformationDes(joint_des_state,&base_state,&base_orient,endeff,
982  joint_cog_mpos_des,joint_axis_pos_des,joint_origin_pos_des,
983  link_pos_des,Alink_des,Adof_des);
984 
985  /* the desired endeffector information */
986  for (i=1; i<=N_CART; ++i) {
987  for (j=1; j<=n_endeffs; ++j) {
988  cart_des_state[j].x[i] = link_pos_des[link2endeffmap[j]][i];
989  }
990  }
991 
992  /* the desired quaternian of the endeffector */
993  for (j=1; j<=n_endeffs; ++j) {
994  linkQuat(Alink_des[link2endeffmap[j]],&(cart_des_orient[j]));
995  }
996 
997 
998  /* addititional link information */
999  linkInformation(joint_state,&base_state,&base_orient,endeff,
1000  joint_cog_mpos,joint_axis_pos,joint_origin_pos,
1001  link_pos,Alink,Adof);
1002 
1003  /* create the endeffector information */
1004  for (i=1; i<=N_CART; ++i) {
1005  for (j=1; j<=n_endeffs; ++j) {
1006  cart_state[j].x[i] = link_pos[link2endeffmap[j]][i];
1007  }
1008  }
1009 
1010  /* the quaternian of the endeffector */
1011  for (j=1; j<=n_endeffs; ++j) {
1012  linkQuat(Alink[link2endeffmap[j]],&(cart_orient[j]));
1013  }
1014 
1015  /* the COG position */
1016  compute_cog();
1017 
1018  /* the jacobian */
1019  jacobian(link_pos,joint_origin_pos,joint_axis_pos,J);
1020  baseJacobian(link_pos,joint_origin_pos,joint_axis_pos,Jbase);
1021 
1022  jacobian(link_pos_des,joint_origin_pos_des,joint_axis_pos_des,Jdes);
1023  baseJacobian(link_pos_des,joint_origin_pos_des,joint_axis_pos_des,Jbasedes);
1024 
1025  /* numerical time derivative of Jacobian */
1026  if (!firsttime) {
1027  mat_sub(J,last_J,dJdt);
1028  mat_mult_scalar(dJdt,(double)task_servo_rate,dJdt);
1029  mat_sub(Jbase,last_Jbase,dJbasedt);
1030  mat_mult_scalar(dJbasedt,(double)task_servo_rate,dJbasedt);
1031  }
1032  mat_equal(J,last_J);
1033  mat_equal(Jbase,last_Jbase);
1034 
1035  /* compute the cartesian velocities and accelerations */
1036 
1037  for (j=1; j<=n_endeffs; ++j) {
1038 
1039  for (i=1; i<=N_CART; ++i) {
1040 
1041  cart_state[j].xd[i] = 0.0;
1042  cart_state[j].xdd[i] = 0.0;
1043 
1044  cart_orient[j].ad[i] = 0.0;
1045  cart_orient[j].add[i] = 0.0;
1046 
1047  cart_des_state[j].xd[i] = 0.0;
1048  cart_des_orient[j].ad[i] = 0.0;
1049

1051  for (r=1; r<=n_dofs; ++r) {
1052  cart_state[j].xd[i] += J[(j-1)*6+i][r] * joint_state[r].thd;
1053  cart_orient[j].ad[i] += J[(j-1)*6+i+3][r] * joint_state[r].thd;
1054 
1055  cart_des_state[j].xd[i] += Jdes[(j-1)*6+i][r] *joint_des_state[r].thd;
1056  cart_des_orient[j].ad[i]+= Jdes[(j-1)*6+i+3][r] * joint_des_state[r].thd;
1057 
1058  cart_state[j].xdd[i] += J[(j-1)*6+i][r] * joint_state[r].thdd +
1059  dJdt[(j-1)*6+i][r] * joint_state[r].thd;
1060  cart_orient[j].add[i] += J[(j-1)*6+i+3][r] * joint_state[r].thdd +
1061  dJdt[(j-1)*6+i+3][r] * joint_state[r].thd;
1062  }
1063 
1064  /* contributations from the base */
1065  for (r=1; r<=N_CART; ++r) {
1066  cart_state[j].xd[i] += Jbase[(j-1)*6+i][r] * base_state.xd[r];
1067  cart_orient[j].ad[i] += Jbase[(j-1)*6+i+3][r] * base_state.xd[r];
1068 
1069  cart_state[j].xd[i] += Jbase[(j-1)*6+i][3+r] * base_orient.ad[r];
1070  cart_orient[j].ad[i] += Jbase[(j-1)*6+i+3][3+r] * base_orient.ad[r];
1071 
1072  cart_des_state[j].xd[i] += Jbasedes[(j-1)*6+i][r] * base_state.xd[r];
1073  cart_des_orient[j].ad[i] += Jbasedes[(j-1)*6+i+3][r] * base_state.xd[r];
1074 
1075  cart_des_state[j].xd[i] += Jbasedes[(j-1)*6+i][3+r] * base_orient.ad[r];
1076  cart_des_orient[j].ad[i] += Jbasedes[(j-1)*6+i+3][3+r] * base_orient.ad[r];
1077 
1078  cart_state[j].xdd[i] += Jbase[(j-1)*6+i][r] * base_state.xdd[r] +
1079  dJbasedt[(j-1)*6+i][r] * base_state.xd[r];
1080  cart_orient[j].add[i] += Jbase[(j-1)*6+i+3][r] * base_state.xdd[r] +
1081  dJbasedt[(j-1)*6+i+3][r] * base_state.xd[r];
1082 
1083  cart_state[j].xdd[i] += Jbase[(j-1)*6+i][3+r] * base_orient.add[r] +
1084  dJbasedt[(j-1)*6+i][3+r] * base_orient.ad[r];
1085  cart_orient[j].add[i] += Jbase[(j-1)*6+i+3][3+r] * base_orient.add[r] +
1086  dJbasedt[(j-1)*6+i+3][3+r] * base_orient.ad[r];
1087  }
1088 
1089  }
1090 
1091  /* compute quaternion derivatives */
1092  quatDerivatives(&(cart_orient[j]));
1093  quatDerivatives(&(cart_des_orient[j]));
1094  for (r=1; r<=N_QUAT; ++r)
1095  cart_des_orient[j].qdd[r] = 0.0; // we don't have dJdes_dt so far
1096 
1097  }
1098 
1099  /* reset first time flag */
1100  firsttime = FALSE;
